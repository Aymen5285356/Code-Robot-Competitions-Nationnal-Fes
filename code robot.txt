#include <Arduino.h>
#include <WiFi.h>
#include <HTTPSServer.hpp>
#include <SSLCert.hpp>
#include <HTTPRequest.hpp>
#include <HTTPResponse.hpp>
#include "esp_camera.h"

// --- Hardware Pins ---
#define IN1 14    // Motor Driver L-Forward
#define IN2 15    // Motor Driver L-Backward
#define IN3 2     // Motor Driver R-Forward
#define IN4 4     // Motor Driver R-Backward
#define TRIG 13   // HC-SR04 Trig
#define ECHO 12   // HC-SR04 Echo
#define IR_L 32   // Left Line Sensor
#define IR_R 33   // Right Line Sensor

using namespace httpsserver;

// --- Global Variables ---
SSLCert * cert;
HTTPSServer * secureServer;
bool autonomousMode = true;

// --- Competition Logic Helper ---
void stopRobot() {
  digitalWrite(IN1, LOW); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, LOW);
}

void moveForward(int speed) {
  analogWrite(IN1, speed); digitalWrite(IN2, LOW);
  analogWrite(IN3, speed); digitalWrite(IN4, LOW);
}

// Ultrasonic check for "Visitor" figurine (10cm requirement) 
bool checkVisitor() {
  digitalWrite(TRIG, LOW); delayMicroseconds(2);
  digitalWrite(TRIG, HIGH); delayMicroseconds(10);
  digitalWrite(TRIG, LOW);
  long duration = pulseIn(ECHO, HIGH);
  float distance = duration * 0.034 / 2;
  return (distance > 0 && distance < 12); // Detect within 12cm
}

// --- HTTPS Handler ---
void handleStatus(HTTPRequest * req, HTTPResponse * res) {
  res->setStatusCode(200);
  res->setHeader("Content-Type", "text/html");
  res->println("<h1>DromAI Status: Online</h1>");
  res->println("<p>Mission: Guide Touristique Maroc</p>");
}

void setup() {
  Serial.begin(115200);
  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);
  pinMode(TRIG, OUTPUT); pinMode(ECHO, INPUT);
  pinMode(IR_L, INPUT);  pinMode(IR_R, INPUT);

  // WiFi Setup
  WiFi.begin("Your_SSID", "Your_PASSWORD");
  while (WiFi.status() != WL_CONNECTED) { delay(500); }

  // HTTPS Cert Setup
  cert = new SSLCert();
  createSelfSignedCert(*cert, KEYSIZE_2048, "CN=dromai.local");
  secureServer = new HTTPSServer(cert);

  ResourceNode * nodeStatus = new ResourceNode("/status", "GET", &handleStatus);
  secureServer->registerNode(nodeStatus);
  secureServer->start();
}

void loop() {
  secureServer->loop();

  if (autonomousMode) {
    // 1. Line Following [cite: 25]
    bool left = digitalRead(IR_L);
    bool right = digitalRead(IR_R);

    if (left == LOW && right == LOW) {
      moveForward(180); // Base speed to handle sand/bumps [cite: 50, 96]
    } else if (left == HIGH) {
      // Turn left
      analogWrite(IN1, 0); analogWrite(IN3, 150);
    } else if (right == HIGH) {
      // Turn right
      analogWrite(IN1, 150); analogWrite(IN3, 0);
    }

    // 2. City Interaction Check (Fès/Laâyoune) [cite: 58, 91, 107]
    if (checkVisitor()) {
      stopRobot();
      Serial.println("Visitor found. Reading QR Code..."); 
      delay(5000); // 5-second display requirement [cite: 66, 88]
    }
  }
}
